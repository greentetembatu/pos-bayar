      let dataMember = JSON.parse(localStorage.getItem("member")) || [];

      function simpanDataToko() {
        let data = {
          tokoNama: document.getElementById("tokoNama").value,
          tokoAlamat: document.getElementById("tokoAlamat").value,
          tokoHp: document.getElementById("tokoHp").value,
          kasir: document.getElementById("kasir").value,
        };
        localStorage.setItem("dataToko", JSON.stringify(data));
        alert("Konfigurasi toko berhasil diperbarui!");
      }

      function generateNo() {
        let now = new Date();
        let tgl = now.getDate().toString().padStart(2, "0");
        let bln = (now.getMonth() + 1).toString().padStart(2, "0");
        let thn = now.getFullYear().toString().slice(-2);
        let urut = (dataMember.length + 1).toString().padStart(4, "0");
        return `MEM-${tgl}${bln}${thn}${urut}`;
      }

      function simpanData() {
        const inpNama = document.getElementById("nama");
        const inpHp = document.getElementById("hp");

        if (!inpNama.value || !inpHp.value) return alert("Isi nama dan HP!");

        let dataToko = JSON.parse(localStorage.getItem("dataToko")) || {};

        let obj = {
          tanggal: new Date().toLocaleDateString("id-ID"),
          kasir: dataToko.kasir || "-",
          nama: inpNama.value,
          hp: inpHp.value,
          no: generateNo(),
        };

        dataMember.push(obj);
        localStorage.setItem("member", JSON.stringify(dataMember));

        tampilKartu(obj);
        
        // Reset form
        inpNama.value = "";
        inpHp.value = "";
      }

      function tampilKartu(d) {
        let dataToko = JSON.parse(localStorage.getItem("dataToko")) || {};
        const kartuEl = document.getElementById("kartu");
        const btnPrint = document.getElementById("btnPrint");
        const barcodeEl = document.getElementById("barcode");

        kartuEl.style.display = "block";
        btnPrint.style.display = "inline-block";

        document.getElementById("cardToko").innerText = dataToko.tokoNama || "NAMA TOKO";
        document.getElementById("cardAlamat").innerText = dataToko.tokoAlamat || "Alamat belum diatur";
        document.getElementById("cardHpToko").innerText = "Telp: " + (dataToko.tokoHp || "-");

        document.getElementById("cardNama").innerText = "Nama: " + d.nama;
        document.getElementById("cardHp").innerText = "WhatsApp: " + d.hp;
        document.getElementById("cardNo").innerText = "ID: " + d.no;

        JsBarcode("#barcode", d.no, {
          format: "CODE128",
          width: 2,
          height: 40,
          displayValue: false,
          background: "#ffffff"
        });
      }

function downloadCSV() {
        if (dataMember.length === 0) return alert("Belum ada data member");
        
        let csv = "Tanggal;Kasir;Nama Member;No HP;No Member\n";
        dataMember.forEach((d) => {
          csv += `${d.tanggal};${d.kasir};${d.nama};'${d.hp};${d.no}\n`;
        });

        // Membuat format tanggal DD-MM-YYYY
        let skrg = new Date();
        let tgl = skrg.getDate().toString().padStart(2, "0");
        let bln = (skrg.getMonth() + 1).toString().padStart(2, "0");
        let thn = skrg.getFullYear();
        let formatTanggal = `${tgl}-${bln}-${thn}`;

        let blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
        let link = document.createElement("a");
        link.href = URL.createObjectURL(blob);
        
        // Nama file baru sesuai permintaan Anda
        link.download = `data-member-${formatTanggal}.csv`;
        
        link.click();
      }
      
function cariData() {
  let key = document.getElementById("search").value.toLowerCase();
  if (!key) return alert("Masukkan kata kunci (Nama, ID, atau No HP)");
  
  let found = dataMember.find((d) => 
    d.nama.toLowerCase().includes(key) || // Cari berdasarkan Nama
    d.no.toLowerCase().includes(key) ||   // Cari berdasarkan ID Member
    d.hp.includes(key)                    // Cari berdasarkan Nomor HP
  );

  if (found) {
    tampilKartu(found);
  } else {
    alert("Member tidak ditemukan!");
  }
}




      let scanner = null;
      function startScanner() {
        if (scanner) return;
        scanner = new Html5Qrcode("reader");
        scanner.start(
          { facingMode: "environment" },
          { fps: 10, qrbox: 250 },
          (decodedText) => {
            new Audio("https://www.soundjay.com/button/sounds/beep-07.mp3").play();
            let found = dataMember.find((d) => d.no === decodedText);
            if (found) {
                tampilKartu(found);
            } else {
                alert("Barcode tidak terdaftar: " + decodedText);
            }
            scanner.stop().then(() => { scanner = null; });
          }
        ).catch(err => alert("Kamera bermasalah"));
      }

      window.onload = function () {
        let data = JSON.parse(localStorage.getItem("dataToko"));
        if (data) {
          document.getElementById("tokoNama").value = data.tokoNama || "";
          document.getElementById("tokoAlamat").value = data.tokoAlamat || "";
          document.getElementById("tokoHp").value = data.tokoHp || "";
          document.getElementById("kasir").value = data.kasir || "";
        }
      };
function printKartu() {
    // Ambil data dari elemen preview kartu
    const cardNo = document.getElementById("cardNo").innerText.replace("ID: ", "");
    const cardToko = document.getElementById("cardToko").innerText;
    const cardAlamat = document.getElementById("cardAlamat").innerText;
    const cardHpToko = document.getElementById("cardHpToko").innerText;
    const cardNama = document.getElementById("cardNama").innerText.replace("Nama: ", "");
    const cardHp = document.getElementById("cardHp").innerText.replace("WhatsApp: ", "");

    if (!cardNo) {
        alert("Pilih atau buat kartu member terlebih dahulu!");
        return;
    }

    // Susun parameter URL untuk dikirim ke cetak.html
    const params = new URLSearchParams({
        no: cardNo,
        toko: cardToko,
        alamat: cardAlamat,
        hpToko: cardHpToko,
        nama: cardNama,
        hp: cardHp
    });

    // Membuka file cetak.html di tab baru dengan membawa data
    window.open("cetak-member.html?" + params.toString(), "_blank");
}
